source ~/ResumeStorageENV/bin/activate
source venv/bin/activate

server {
    listen 80;
    server_name 3.15.253.244;

    # to avoid any error while fetching fevicon
    location = /favicon.ico { access_log off; log_not_found off; }

    location /static/ {
        root /home/ubuntu/CandidateCollector;
    }

    location / {
        include proxy_params;
        # communicate via socket file created by Gunicorn
        proxy_pass http://unix:/home/ubuntu/CandidateCollector/CandidateCollector.sock;
    }
}

[Unit]
Description=Gunicorn daemon for CandidateCollector
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/CandidateCollector
ExecStart=/home/ubuntu/CandidateCollector/env/bin/gunicorn --access-logfile - --workers 3 --bind unix:/home/ubuntu/CandidateCollector/candidatecollector.sock CandidateCollector.wsgi:application

[Install]
WantedBy=multi-user.target
